49# definicion of a grammar for parser
50grammar = """
51 ?start: sum
52 | NAME "=" sum -> assign_var
53 ?sum: product
54 | sum "+" product -> add
55 | sum "-" product -> sub
56 ?product: atom
57 | product "^" atom -> exp
58 | product "*" atom -> mul
59 | product "/" atom -> div
60 ?atom: NUMBER -> number
61 | "&" atom -> ln
62 | atom "$" atom -> root
63 | atom "!" -> fac
64 | "-" atom -> neg
65 | NAME -> var
66 | "(" sum ")"
67 %import common.CNAME -> NAME
68 %import common.NUMBER
69 %import common.WS_INLINE
70 %ignore WS_INLINE
71"""

98# Create parser from Lark library. This parser is using grammar defined prieviously.

104def calculate(data) -> None:
105 data = data.replace(chr(0x221A), '$')
106 data = data.replace("ln", '&')
107 calc_parser = Lark(grammar, parser='lalr', transformer=CalculateTree())
108 calc = calc_parser.parse
109 try:
110 result = calc(data)
111 return result
